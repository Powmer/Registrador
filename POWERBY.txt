import pandas as pd
import matplotlib.pyplot as plt


dataset['Preço'] = pd.to_numeric(dataset['Preço'], errors='coerce')


soma_preco = dataset['Preço'].sum()
media_preco = soma_preco.mean()
plt.figure(figsize=(6, 4))
plt.axis('off')

import pandas as pd
